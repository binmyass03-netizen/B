--[[
    Brainrot Animation Controller - PODIUMS ONLY
    
    A collapsible and draggable popup to control animations on brainrots in your plot.
    
    Features:
    - Control brainrots on your plot (click numbered buttons to select)
    - SELECT ALL to control all animals at once
    - Lists all available animations for the brainrot
    - Play, stop any animation with looping
    - Speed control slider
    - Works with both real game animals AND client-spawned animals
    - Auto-refresh with heartbeat (detects when animals change)
    
    Usage:
    - Click the 🎬 toggle button (draggable) to open
    - Click a numbered podium button to select that brainrot
    - Or click "ALL" to select all brainrots
    - Click any animation to play it
]] -- ============================================================================

    local function getPodiums()
        local plot = getMyPlot()

        if not folder then

    -- Get model EXACTLY like game does:
    -- PODIUM ANIMALS: PlotClient creates models parented to Plot, positioned at AnimalPodiums>[index]>Base>Spawn
    -- The model NAME matches the animal Index (e.g. "Tralalero Tralala")
    local function getAnimalOnPodium(podiumIndex)
        local plot = getMyPlot()
        if not plot then
            return nil, nil
        end

        local channel = getPlotChannel()
        if not channel then
            return nil, nil
        end

        -- Get animal data from channel's AnimalList
        local data
        pcall(function()
            data = channel:Get(string.format("AnimalList.%d", podiumIndex))
        end)

        if not data or data == "Empty" then
            pcall(function()
                local animalList = channel:Get("AnimalList")
                if animalList and type(animalList) == "table" then
                    data = animalList[podiumIndex] or animalList[tostring(podiumIndex)]
                end
            end)
        end

        if not data or data == "Empty" or type(data) ~= "table" or not data.Index then
            return nil, nil
        end

        -- Get the podium spawn point position
        local podiumsFolder = plot:FindFirstChild("AnimalPodiums") or plot:FindFirstChild("Podiums")
        if not podiumsFolder then
            return nil, data
        end

        local podiumFolder = podiumsFolder:FindFirstChild(tostring(podiumIndex))
        if not podiumFolder then
            return nil, data
        end

        local base = podiumFolder:FindFirstChild("Base")
        local spawnPart = base and base:FindFirstChild("Spawn")
        if not spawnPart then
            return nil, data
        end

        local spawnPos = spawnPart.Position

        -- GAME LOGIC: PlotClient parents the animal model DIRECTLY to the plot
        -- The model's name matches the animal Index from AnimalList data
        local model = nil
        local closestDist = 15 -- Must be within 15 studs of spawn point

        -- Check direct children of plot
        for _, child in ipairs(plot:GetChildren()) do
            if child:IsA("Model") and child.Name == data.Index then
                local primary = child.PrimaryPart or child:FindFirstChildWhichIsA("BasePart")
                if primary then
                    local dist = (primary.Position - spawnPos).Magnitude
                    if dist < closestDist then
                        closestDist = dist
                        model = child
                    end
                end
            end
        end

        -- Fallback: search descendants if not found
        if not model then
            for _, child in ipairs(plot:GetDescendants()) do
                if child:IsA("Model") and child.Name == data.Index then
                    local primary = child.PrimaryPart or child:FindFirstChildWhichIsA("BasePart")
                    if primary then
                        local dist = (primary.Position - spawnPos).Magnitude
                        if dist < closestDist then
                            closestDist = dist
                            model = child
                        end
                    end
                end
            end
        end

        return model, data
    end

    local function getAvailableAnimations(animalName)
        local folder = AnimalAnimations:FindFirstChild(animalName)
        if not folder then
            return {}
        end
        local anims = {}
        for _, child in ipairs(folder:GetChildren()) do
            if child:IsA("Animation") then
                table.insert(anims, {
                    name = child.Name,
                    id = child.AnimationId
                })
            end
        end
        table.sort(anims, function(a, b)
            if a.name == "Idle" then
                return true
            end
            if b.name == "Idle" then
                return false
            end
            return a.name < b.name
        end)
        return anims
    end

    local function setSpeed(speed)
        -- Adjust speed of current playing track
        if currentTrack then
            pcall(function()
                currentTrack:AdjustSpeed(speed)
            end)
        end
        -- Also adjust for podium animators in SELECT ALL mode
        for _, entry in ipairs(podiumAnimators) do
            if entry.currentTrack then
                pcall(function()
                    entry.currentTrack:AdjustSpeed(speed)
                end)
            end
        end
    end

    -- Track our custom animations and the original tracks we stopped
    local currentTrack = nil
    local savedOriginalTracks = {} -- Save game's tracks so we can restore them

    local function playAnimation(animalName, animName, animator)
        local targetAnimator = animator or currentAnimator
        print("[AnimController] playAnimation:", animalName, animName, "animator:", targetAnimator and "OK" or "NIL")
        if not targetAnimator then
            warn("[AnimController] No animator!")
            return false
        end

        local folder = AnimalAnimations:FindFirstChild(animalName)
        if not folder then
            warn("[AnimController] No animation folder for:", animalName)
            return false
        end

        local anim = folder:FindFirstChild(animName)
        if not anim then
            warn("[AnimController] No animation:", animName, "for", animalName)
            return false
        end

        local animatorKey = tostring(targetAnimator)

        -- Stop our current custom track if playing
        if currentTrack then
            pcall(function()
                currentTrack:Stop(0.2)
            end)
            currentTrack = nil
        end

        -- Save and stop all currently playing tracks (game's animations)
        if not savedOriginalTracks[animatorKey] then
            savedOriginalTracks[animatorKey] = {}
            pcall(function()
                for _, track in ipairs(targetAnimator:GetPlayingAnimationTracks()) do
                    table.insert(savedOriginalTracks[animatorKey], track)
                    track:Stop(0.2)
                end
            end)
        else
            -- Already saved, just stop any playing
            pcall(function()
                for _, track in ipairs(targetAnimator:GetPlayingAnimationTracks()) do
                    track:Stop(0.2)
                end
            end)
        end

        -- Load and play our animation
        local track
        local ok = pcall(function()
            track = targetAnimator:LoadAnimation(anim)
            track.Looped = true
            track:Play(0.2)
            track:AdjustSpeed(currentSpeed)
        end)

        if ok and track then
            currentTrack = track
            currentPlayingTrack = track
            currentPlayingAnimName = animName
            return true
        end

        warn("[AnimController] Failed to play animation")
        return false
    end

    local function stopAllAnimations()
        -- Stop our custom track (single mode)
        if currentTrack then
            pcall(function()
                currentTrack:Stop(0.2)
            end)
            currentTrack = nil
        end

        -- Stop all currently playing tracks (game's or custom) for single mode
        for animatorKey, tracks in pairs(savedOriginalTracks) do
            for _, track in ipairs(tracks) do
                pcall(function()
                    track:Stop(0.2)
                end)
            end
        end
        savedOriginalTracks = {}

        -- Stop all tracks for SELECT ALL mode (podiumAnimators)
        for _, entry in ipairs(podiumAnimators) do
            pcall(function()
                -- Stop custom track
                if entry.currentTrack then
                    entry.currentTrack:Stop(0.2)
                    entry.currentTrack = nil
                end
                -- Stop all saved original tracks
                if entry.savedTracks then
                    for _, track in ipairs(entry.savedTracks) do
                        pcall(function()
                            track:Stop(0.2)
                        end)
                    end
                    entry.savedTracks = nil
                end
            end)
        end

        currentPlayingTrack = nil
        currentPlayingAnimName = nil
    end

    local function setupAnimator(model)
        if not model then
            return nil
        end

        -- First check if model has a Humanoid with Animator (common pattern)
        local humanoid = model:FindFirstChildOfClass("Humanoid")
        if humanoid then
            local animator = humanoid:FindFirstChildOfClass("Animator")
            if animator then
                return animator
            end
        end

        -- Then check for AnimationController (alternate pattern)
        local ac = model:FindFirstChildOfClass("AnimationController")
        if ac then
            local animator = ac:FindFirstChildOfClass("Animator")
            if animator then
                return animator
            end
            -- AnimationController exists but no Animator, create one
            local ok, newAnim = pcall(function()
                local a = Instance.new("Animator")
                a.Parent = ac
                return a
            end)
            if ok then
                return newAnim
            end
        end

        -- No existing animation setup, create AnimationController + Animator
        local ok, newAC = pcall(function()
            local c = Instance.new("AnimationController")
            c.Parent = model
            return c
        end)
        if not ok then
            return nil
        end
        ac = newAC

        local ok2, newAnim = pcall(function()
            local a = Instance.new("Animator")
            a.Parent = ac
            return a
        end)
        if ok2 then
            return newAnim
        end

        return nil
    end

    local currentPodiumIndex = nil

    local function selectPodium(podiumIndex, onSelectionMade)
        -- Don't stop animations when just selecting - only reset our tracking
        currentTrack = nil
        savedOriginalTracks = {}
        currentAnimator = nil
        currentAnimalName = nil
        currentPodiumIndex = podiumIndex
        currentPlayingTrack = nil
        currentPlayingAnimName = nil

        if not podiumIndex then
            return nil, nil
        end

        local model, data = getAnimalOnPodium(podiumIndex)

        if model and data then
            currentAnimalName = data.Index
            currentAnimator = setupAnimator(model)
            currentModel = model
        end

        if onSelectionMade then
            onSelectionMade(model, data)
        end
        return model, data
    end

    local function podiumHasAnimal(podiumIndex)
        local model, data = getAnimalOnPodium(podiumIndex)
        return model ~= nil, model, data

    -- Get all brainrots on player's plot (from Synchronizer data)
    local function getBrainrotsOnMyPlot()
        local results = {}
        local channel = getPlotChannel()
        if not channel then
            return results
        end

        local animalList
        pcall(function()
            animalList = channel:Get("AnimalList")
        end)

        if not animalList or type(animalList) ~= "table" then
            return results
        end

        local podiums = getPodiums()
        for _, p in ipairs(podiums) do
            local model, data = getAnimalOnPodium(p.index)
            if data and data.Index and model then
                table.insert(results, {
                    podiumIndex = p.index,
                    model = model,
                    index = data.Index,
                    mutation = data.Mutation,
                    traits = data.Traits,
                    displayName = AnimalsData[data.Index] and AnimalsData[data.Index].DisplayName or data.Index
                })
            end
        end

        return results
    end

            if #anims == 0 then
                local noLbl = Instance.new("TextLabel")
                noLbl.Size = UDim2.new(1, -8, 0, 30)
                noLbl.BackgroundTransparency = 1
                noLbl.Text = "No animations for " .. animalName

                noLbl.Font = Enum.Font.GothamBold
                noLbl.ZIndex = 303
                noLbl.Parent = animScroll
                return
            end

            for i, anim in ipairs(anims) do
                local btn = Instance.new("TextButton")
                btn.Name = "Anim_" .. anim.name
                btn.Size = UDim2.new(1, -8, 0, 34)
                btn.LayoutOrder = i
                btn.BackgroundColor3 = Color3.fromRGB(60, 50, 80)
                btn.BorderSizePixel = 0
                btn.Text = "▶️ " .. anim.name
                btn.TextColor3 = Color3.new(1, 1, 1)
                btn.TextSize = 13
                btn.Font = Enum.Font.GothamBold
                btn.ZIndex = 303
                btn.Parent = animScroll
                createCorner(btn, 6)

                animButtons[anim.name] = btn

                btn.MouseButton1Click:Connect(function()
                    for _, b in pairs(animButtons) do
                        b.BackgroundColor3 = Color3.fromRGB(60, 50, 80)
                    end

                    if isSelectAllMode and #podiumAnimators > 0 then
                        -- Play on all animators (with track saving/restore)
                        local successCount = 0
                        print("[AnimController] Playing", anim.name, "on", #podiumAnimators, "animators")
                        for _, entry in ipairs(podiumAnimators) do
                            local folder = AnimalAnimations:FindFirstChild(entry.animalName)
                            print("[AnimController]   Entry:", entry.animalName, "folder:", folder and "OK" or "NIL",
                                "animator:", entry.animator and "OK" or "NIL")
                            if folder and entry.animator then
                                local animObj = folder:FindFirstChild(anim.name)
                                if not animObj then
                                    for _, child in ipairs(folder:GetChildren()) do
                                        if child:IsA("Animation") and child.Name:lower() == anim.name:lower() then
                                            animObj = child
                                            break
                                        end
                                    end
                                end
                                if not animObj and (anim.name == "Idle" or anim.name == "Walk") then
                                    animObj = folder:FindFirstChildOfClass("Animation")
                                end

                                print("[AnimController]     animObj:", animObj and animObj.Name or "NIL")
                                if animObj then
                                    pcall(function()
                                        local animatorKey = tostring(entry.animator)

                                        -- Stop previous custom track if any
                                        if entry.currentTrack then
                                            entry.currentTrack:Stop(0.2)
                                        end

                                        -- Save original tracks (game's idle) first time only
                                        if not entry.savedTracks then
                                            entry.savedTracks = {}
                                            for _, t in ipairs(entry.animator:GetPlayingAnimationTracks()) do
                                                table.insert(entry.savedTracks, t)
                                                t:Stop(0.2)
                                            end
                                        else
                                            -- Already saved, just stop playing tracks
                                            for _, t in ipairs(entry.animator:GetPlayingAnimationTracks()) do
                                                t:Stop(0.2)
                                            end
                                        end

                                        local track = entry.animator:LoadAnimation(animObj)
                                        track.Looped = true
                                        track:Play(0.2)
                                        track:AdjustSpeed(currentSpeed)
                                        entry.currentTrack = track
                                        successCount = successCount + 1
                                    end)
                                end
                            end
                        end
                        btn.Text = "▶️ " .. anim.name .. " (" .. successCount .. "/" .. #podiumAnimators .. ")"
                        if successCount > 0 then
                            btn.BackgroundColor3 = ACCENT_COLOR
                        else
                            btn.BackgroundColor3 = Color3.fromRGB(200, 80, 80)
                        end
                        task.delay(2, function()
                            if btn.Parent then
                                btn.Text = "▶️ " .. anim.name
                            end
                        end)
                    else
                        -- Single mode
                        if playAnimation(currentAnimalName, anim.name) then
                            btn.BackgroundColor3 = ACCENT_COLOR
                        else
                            btn.BackgroundColor3 = Color3.fromRGB(200, 80, 80)
                            task.delay(0.5, function()
                                if btn.Parent then
                                    btn.BackgroundColor3 = Color3.fromRGB(60, 50, 80)
                                end
                            end)
                        end
                    end
                end)
            end
        end

        -- ========================================================================
        -- HEARTBEAT AUTO-REFRESH (updates podium list every 2 seconds)
        -- ========================================================================
        local lastRefreshTime = 0
        local REFRESH_INTERVAL = 2 -- seconds
        local lastPodiumSnapshot = {} -- Track what we last saw

        local function getPodiumSnapshot()
            local snapshot = {}
            local podiums = getPodiums()
            for _, p in ipairs(podiums) do
                local hasAnimal = podiumHasAnimal(p.index)
                snapshot[p.index] = hasAnimal
            end
            return snapshot
        end

        local function snapshotsMatch(a, b)
            for k, v in pairs(a) do
                if b[k] ~= v then
                    return false
                end
            end
            for k, v in pairs(b) do
                if a[k] ~= v then
                    return false
                end
            end
            return true
        end

        local heartbeatConn
        heartbeatConn = RunService.Heartbeat:Connect(function()
            if not screenGui or not screenGui.Parent then
                if heartbeatConn then
                    heartbeatConn:Disconnect()
                end
                return
            end

            -- Only refresh when UI is visible
            if not screenGui.Enabled or isCollapsed then
                return
            end

            local now = tick()
            if now - lastRefreshTime < REFRESH_INTERVAL then
                return
            end
            lastRefreshTime = now

            -- Check if podium state changed
            local newSnapshot = getPodiumSnapshot()
            if not snapshotsMatch(lastPodiumSnapshot, newSnapshot) then
                lastPodiumSnapshot = newSnapshot
                updatePodiumButtons(false) -- Don't auto-select, just refresh buttons
            end
        end)

        mainUI = {
            screenGui = screenGui,
            panel = panel,
            refreshPodiums = function()
                updatePodiumButtons(false)
            end
        }

        return mainUI
    end

        toggleBtn.AnchorPoint = Vector2.new(0, 0)

        local savedPos = loadConfig(TOGGLE_FILE)
        if savedPos then
            toggleBtn.Position = UDim2.new(0, savedPos.X or 10, 0, savedPos.Y or 200)
        else
            toggleBtn.Position = UDim2.new(0, 10, 0, 300)
        end

        toggleBtn.BackgroundColor3 = HEADER_COLOR
        toggleBtn.BorderSizePixel = 0
        toggleBtn.Text = "🎬"
        toggleBtn.TextSize = 24
        toggleBtn.ZIndex = 100
        toggleBtn.Parent = screenGui
        createCorner(toggleBtn, 10)
        createStroke(toggleBtn, ACCENT_COLOR, 2)

        local dragging, dragStart, startPos = false, nil, nil
        local dragThreshold = 5
        local wasDragging = false

        toggleBtn.InputBegan:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 then
                dragging = true
                wasDragging = false
                dragStart = input.Position
                startPos = toggleBtn.AbsolutePosition
            end
        end)

        local function saveTogglePos()
            local x = toggleBtn.AbsolutePosition.X
            local y = toggleBtn.AbsolutePosition.Y
            saveConfig(TOGGLE_FILE, {
                X = x,
                Y = y
            })
        end

        toggleBtn.InputEnded:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 then
                if not wasDragging then
                    if not mainUI then
                        createMainUI()
                    else
                        mainUI.screenGui.Enabled = not mainUI.screenGui.Enabled
                    end
                else
                    -- Save using AbsolutePosition for accuracy
                    task.defer(saveTogglePos)
                end
                dragging = false
                wasDragging = false
            end
        end)

        UserInputService.InputChanged:Connect(function(input)
            if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
                local delta = input.Position - dragStart
                if delta.Magnitude > dragThreshold then
                    wasDragging = true
                end
                if wasDragging then
                    toggleBtn.Position = UDim2.new(0, startPos.X + delta.X, 0, startPos.Y + delta.Y)
                end
            end
        end)

        return toggleBtn
    end

    -- ============================================================================
    -- GLOBAL API & SIGNAL SYSTEM
    -- ============================================================================

    -- Create BindableEvent for spawn notifications from Index V2
    local AnimalSpawnedEvent = Instance.new("BindableEvent")
    AnimalSpawnedEvent.Name = "AnimalSpawnedEvent"

    -- Auto-animation settings
    local autoAnimEnabled = false
    local autoAnimName = "Idle" -- Default animation to auto-play on spawn

    -- Handler for when Index V2 spawns an animal
    local function onAnimalSpawned(animalName, podiumIndex, mutation)
        if not autoAnimEnabled then
            return
        end

        print("[AnimController] 🎬 Auto-anim triggered:", animalName, "on podium", podiumIndex)

        -- Wait a moment for the model to spawn
        task.delay(0.5, function()
            local model, data = getAnimalOnPodium(podiumIndex)
            if model then
                local animator = setupAnimator(model)
                if animator then
                    -- Play the auto animation
                    local folder = AnimalAnimations:FindFirstChild(animalName)
                    if folder then
                        local animObj = folder:FindFirstChild(autoAnimName)
                        if not animObj then
                            -- Try case-insensitive match
                            for _, child in ipairs(folder:GetChildren()) do
                                if child:IsA("Animation") and child.Name:lower() == autoAnimName:lower() then
                                    animObj = child
                                    break
                                end
                            end
                        end
                        if animObj then
                            pcall(function()
                                -- Stop game's animations
                                for _, track in ipairs(animator:GetPlayingAnimationTracks()) do
                                    track:Stop(0.2)
                                end
                                -- Play our animation
                                local track = animator:LoadAnimation(animObj)
                                track.Looped = true
                                track:Play(0.2)
                                track:AdjustSpeed(currentSpeed)
                                print("[AnimController] ✅ Auto-played", autoAnimName, "on", animalName)
                            end)
                        end
                    end
                end
            end
        end)
    end

    AnimalSpawnedEvent.Event:Connect(onAnimalSpawned)

    _G.BrainrotAnimController = {
        show = function()
            if not mainUI then
                createMainUI()
            else
                mainUI.screenGui.Enabled = true
            end
        end,
        hide = function()
            if mainUI then
                mainUI.screenGui.Enabled = false
            end
        end,
        toggle = function()
            if not mainUI then
                createMainUI()
            else
                mainUI.screenGui.Enabled = not mainUI.screenGui.Enabled
            end
        end,
        -- Manual refresh podium buttons
        refresh = function()
            if mainUI and mainUI.refreshPodiums then
                mainUI.refreshPodiums()
            end
        end,
        playAnimation = playAnimation,
        stopAll = stopAllAnimations,
        setSpeed = setSpeed,
        getAnimations = getAvailableAnimations,
        getBrainrotsOnPlot = getBrainrotsOnMyPlot,

        -- Signal for Index V2 to fire when spawning
        onSpawn = AnimalSpawnedEvent,

        -- Fire this from Index V2: _G.BrainrotAnimController.notifySpawn(animalName, podiumIndex, mutation)
        notifySpawn = function(animalName, podiumIndex, mutation)
            AnimalSpawnedEvent:Fire(animalName, podiumIndex, mutation)
        end,

        -- Auto-animation controls
        setAutoAnim = function(enabled, animName)
            autoAnimEnabled = enabled
            if animName then
                autoAnimName = animName
            end
            print("[AnimController] Auto-anim:", enabled and "ON" or "OFF", "Animation:", autoAnimName)
        end,

        getAutoAnim = function()
            return autoAnimEnabled, autoAnimName
        end
    }

    -- Initialize
    print("[AnimController] 🎬 Loaded! Click numbered podium buttons to select brainrots, or use ALL")
    createToggleButton()
end)
